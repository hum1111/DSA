create or replace function fib(n int) returns void as $$

declare i int; t1 int :=0; t2 int :=1; t3 int :=0;

begin raise notice '%' ,t1; raise notice '%' ,t2; for i in 2..n loop t3:=t1+t2; t1:=t2; t2:=t3; raise notice '%' ,t3; end loop; end $$ language plpgsql; select fib(10);

create or replace function prime(n int) returns void as $$

declare i int; f int :=0;

begin if n=2 then f:=0; else for i in 2..(n-1) loop if(mod(n,i)=0) then f:=1; exit; end if; end loop; end if; if(f=0) then raise notice 'prime number'; Else raise notice 'not prime number'; end if; end $$ language plpgsql;

select prime(5); select prime(6);
